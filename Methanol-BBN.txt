local activeLines = {} -- [obj] = {line, text}

local function getLineTarget(obj)
	if typeof(obj) == "Instance" then
		if obj:IsA("BasePart") then
			return obj.Position, obj.CFrame
		elseif obj:IsA("Model") then
			local part = obj.PrimaryPart or obj:FindFirstChildWhichIsA("BasePart")
			if part then return part.Position, part.CFrame end
		end
	elseif typeof(obj) == "Vector3" then
		return obj, CFrame.new(obj)
	end
	return nil, nil
end

local function createLine(obj, config)
	if activeLines[obj] then return end
	
	local line = Drawing.new("Line")
	line.Visible = true
	line.Thickness = 2
	line.Color = config.color
	line.Transparency = 0.2
	
	local text = Drawing.new("Text")
	text.Visible = true
	text.Size = 13
	text.Center = true
	text.Outline = true
	text.OutlineColor = Color3.new(0, 0, 0)
	text.Color = config.color
	text.Text = config.text
	
	activeLines[obj] = {line = line, text = text, config = config}
end

local function removeLine(obj)
	local data = activeLines[obj]
	if data then
		data.line:Remove()
		data.text:Remove()
		activeLines[obj] = nil
	end
end

local function updateLines()
	local cam = workspace.CurrentCamera
	local char = player.Character
	local hrp = char and char:FindFirstChild("HumanoidRootPart")
	if not hrp then return end
	
	for obj, data in pairs(activeLines) do
		local pos, cf = getLineTarget(obj)
		
		-- Check if object still exists
		if not pos or (typeof(obj) == "Instance" and not obj.Parent) then
			removeLine(obj)
		else
			local startPos = hrp.Position + Vector3.new(0, 1, 0) -- Slightly above torso
			local screenStart = cam:WorldToViewportPoint(startPos)
			local screenEnd = cam:WorldToViewportPoint(pos)
			
			-- Only show if target is in front of camera
			local _, onScreen = cam:WorldToViewportPoint(pos)
			local inFront = screenEnd.Z > 0
			
			if onScreen and inFront then
				data.line.Visible = true
				data.line.From = Vector2.new(screenStart.X, screenStart.Y)
				data.line.To = Vector2.new(screenEnd.X, screenEnd.Y)
				
				-- Distance-based opacity
				local dist = (hrp.Position - pos).Magnitude
				local alpha = math.clamp(1 - (dist / 300), 0.3, 1)
				data.line.Transparency = 1 - alpha
				data.line.Color = data.config.color * Color3.new(alpha, alpha, alpha)
				
				-- Text label at target position
				data.text.Visible = true
				data.text.Position = Vector2.new(screenEnd.X, screenEnd.Y - 20)
				data.text.Text = string.format("%s [%dm]", data.config.text, math.floor(dist))
				data.text.Color = data.config.color
			else
				data.line.Visible = false
				data.text.Visible = false
			end
		end
	end
end

local function scanForLines()
	cacheFolders()
	
	-- Generators
	if lineConfig.Generator.enabled then
		for _, v in ipairs(workspace:GetDescendants()) do
			if v.Name == "Generator" and v:IsA("Model") then
				local part = v.PrimaryPart or v:FindFirstChildWhichIsA("BasePart")
				if part and not activeLines[v] then
					createLine(v, lineConfig.Generator)
				end
			end
		end
	end
	
	-- Batteries
	if lineConfig.Battery.enabled then
		for _, v in ipairs(workspace:GetDescendants()) do
			if v.Name == "Battery" and v:IsA("BasePart") then
				if not activeLines[v] then
					createLine(v, lineConfig.Battery)
				end
			end
		end
	end
	
	-- Killer
	if lineConfig.Killer.enabled and CachedFolders.KILLER then
		local killer = CachedFolders.KILLER:GetChildren()[1]
		if killer and not activeLines[killer] then
			createLine(killer, lineConfig.Killer)
		end
	end
	
	-- Survivors
	if lineConfig.Survivor.enabled and CachedFolders.ALIVE then
		for _, v in ipairs(CachedFolders.ALIVE:GetChildren()) do
			if v ~= player.Character and not activeLines[v] then
				createLine(v, lineConfig.Survivor)
			end
		end
	end
end

local function clearAllLines()
	for obj, _ in pairs(activeLines) do
		removeLine(obj)
	end
end

--// CONNECTION STORAGE
local conns = {}
local function safeConnect(name, signal, func)
	if conns[name] then conns[name]:Disconnect() end
	conns[name] = signal:Connect(func)
	return conns[name]
end
local function disconnect(name)
	if conns[name] then conns[name]:Disconnect() conns[name] = nil end
end

local function lerpMove(target)
	local root = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
	if not root then
		notify("Error", "No HumanoidRootPart found")
		return false
	end
	if not target or typeof(target) ~= "CFrame" then
		notify("Error", "Invalid target CFrame")
		return false
	end
	local start, dist = root.CFrame, (root.Position - target.Position).Magnitude
	local dur, t = dist / 25, 0
	while t < dur and root and root.Parent do
		t = t + RunService.Heartbeat:Wait()
		root.CFrame = start:Lerp(target, math.min(t/dur, 1))
	end
	if root and root.Parent then root.CFrame = target end
	return true
end

		atm.Density = 0
		atm.Haze = 0
		atm.Glare = 0
	end
end

local function saveLighting()
	originalLighting = {
		Brightness = Lighting.Brightness,
		ClockTime = Lighting.ClockTime,
		FogEnd = Lighting.FogEnd,
		GlobalShadows = Lighting.GlobalShadows,
		Ambient = Lighting.Ambient,
		OutdoorAmbient = Lighting.OutdoorAmbient
	}
	local atm = Lighting:FindFirstChildOfClass("Atmosphere")
	if atm then
		originalLighting.Atmosphere = atm
		originalLighting.Density = atm.Density
		originalLighting.Haze = atm.Haze
		originalLighting.Glare = atm.Glare
	end
end

local function restoreLighting()
	if not next(originalLighting) then return end
	Lighting.Brightness = originalLighting.Brightness
	Lighting.ClockTime = originalLighting.ClockTime
	Lighting.FogEnd = originalLighting.FogEnd
	Lighting.GlobalShadows = originalLighting.GlobalShadows
	Lighting.Ambient = originalLighting.Ambient
	Lighting.OutdoorAmbient = originalLighting.OutdoorAmbient
	if originalLighting.Atmosphere then
		originalLighting.Atmosphere.Density = originalLighting.Density
		originalLighting.Atmosphere.Haze = originalLighting.Haze
		originalLighting.Atmosphere.Glare = originalLighting.Glare
	end
end

local function doESP(folder, tbl, color, enabled)
	if enabled and folder then

				local h = Instance.new("Highlight", v)

		if folder then disconnect("ESP_"..folder.Name) disconnect("ESPR_"..folder.Name) end
		for v, h in pairs(tbl) do if h then h:Destroy() end end
		table.clear(tbl)
	end
end
