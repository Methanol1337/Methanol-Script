local CachedFolders = {}

local function cacheFolders()
	CachedFolders.PLAYERS = workspace:FindFirstChild("PLAYERS")
	if CachedFolders.PLAYERS then
		CachedFolders.ALIVE = CachedFolders.PLAYERS:FindFirstChild("ALIVE")
		CachedFolders.KILLER = CachedFolders.PLAYERS:FindFirstChild("KILLER")
		CachedFolders.LOBBY = CachedFolders.PLAYERS:FindFirstChild("LOBBY")
	end
	CachedFolders.GAME = workspace:FindFirstChild("GAME")
	CachedFolders.MAPS = workspace:FindFirstChild("MAPS")
	if CachedFolders.MAPS then
		CachedFolders.GAME_MAP = CachedFolders.MAPS:FindFirstChild("GAME MAP")
	end
end

-- Single optimized main loop for lightweight operations
local function MainLoop()
	local currentTime = tick()
	local char = player.Character
	local myRoot = char and char:FindFirstChild("HumanoidRootPart")

	-- Noclip (every frame - needed for physics)
	if ActiveLoops.noclip and char then
		for _, v in pairs(char:GetDescendants()) do
			if v:IsA("BasePart") then
				v.CanCollide = false
			end
		end
	end

	-- Infinite Sprint (every frame - needed for responsiveness)
	if ActiveLoops.infiniteSprint and char then
		local hum = char:FindFirstChildOfClass("Humanoid")
		if CachedFolders.LOBBY and char.Parent == CachedFolders.LOBBY then
			char:SetAttribute("WalkSpeed", 12)
		elseif hum then
			if isMobile then
				if hum.MoveDirection.Magnitude > 0 then
					char:SetAttribute("WalkSpeed", 24)
				else
					char:SetAttribute("WalkSpeed", 12)
				end
			else
				if UIS:IsKeyDown(Enum.KeyCode.LeftShift) or UIS:IsKeyDown(Enum.KeyCode.RightShift) then
					char:SetAttribute("WalkSpeed", 24)
				else
					char:SetAttribute("WalkSpeed", 12)
				end
			end
		end
	end
end

local function StartMainLoop()
	if not MainLoopConnection then
		MainLoopConnection = RunService.Heartbeat:Connect(MainLoop)
	end
end

local conns = {}

--// SAFE CONNECTION WRAPPER (prevents connection leaks)
local function safeConnect(name, signal, func)
	if conns[name] then
		conns[name]:Disconnect()
		conns[name] = nil
	end
	conns[name] = signal:Connect(func)
	return conns[name]
end

local function disconnect(name)
	if conns[name] then
		conns[name]:Disconnect()
		conns[name] = nil
	end
end

local cooldowns = {}
local function throttle(name, cd)
	local now = tick()
	if cooldowns[name] and now - cooldowns[name] < cd then

	end

	return true

--// TOGGLE FUNCTION
local function createToggle(tab, text, default, callback)
	local btn = Instance.new("TextButton", tab)
	btn.Size = UDim2.new(1, -10, 0, 26)
	btn.Font = Enum.Font.Gotham
	btn.TextSize = 18
	local enabled = default or false

	local function update(state)
		enabled = state
		btn.Text = (enabled and "[ON] " or "[OFF] ") .. text
		callback(enabled)
	end

	btn.MouseButton1Click:Connect(function()
		update(not enabled)
	end)

	-- initial state
	task.defer(function()
		update(enabled)
	end)

	return btn
end

local rDragging = false
local rStartPos = nil
local rStart = nil
local DRAG_THRESHOLD = 10
local moved = false

reopenBtn.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 
	or input.UserInputType == Enum.UserInputType.Touch then
		
		rDragging = true
		moved = false
		rStart = input.Position
		rStartPos = reopenBtn.Position
	end
end)

UIS.InputChanged:Connect(function(input)
	if not rDragging then return end
	if input.UserInputType ~= Enum.UserInputType.MouseMovement 
	and input.UserInputType ~= Enum.UserInputType.Touch then return end

	local currentPos = input.Position
	local totalDelta = currentPos - rStart

	-- detect movement
	if not moved and totalDelta.Magnitude > DRAG_THRESHOLD then
		moved = true
	end

	-- move button
	reopenBtn.Position = UDim2.new(
		rStartPos.X.Scale, rStartPos.X.Offset + totalDelta.X,
		rStartPos.Y.Scale, rStartPos.Y.Offset + totalDelta.Y
	)
end)

reopenBtn.InputEnded:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 
	or input.UserInputType == Enum.UserInputType.Touch then
		
		rDragging = false

		-- final safety check
		local endPos = input.Position
		local totalDelta = endPos - rStart

		if totalDelta.Magnitude <= DRAG_THRESHOLD and not moved then
			MainFrame.Visible = true
			reopenBtn.Visible = false
		end

		moved = false
	end
end)

--// AIMLOCK (with bindable key and toggle)
local aimlockEnabled = false
local lockedTarget = nil
local aimlockBind = Enum.UserInputType.MouseButton2

local function getMyTeamFolder()
	local char = player.Character
	if not char then return nil end
	local aliveFolder = workspace:FindFirstChild("PLAYERS") and workspace.PLAYERS:FindFirstChild("ALIVE")
	local killerFolder = workspace:FindFirstChild("PLAYERS") and workspace.PLAYERS:FindFirstChild("KILLER")

	if aliveFolder and char.Parent == aliveFolder then
		return "ALIVE"
	elseif killerFolder and char.Parent == killerFolder then
		return "KILLER"
	end
	return nil
end

createToggle(combatTab, "Aimlock", false, function(enabled)
	aimlockEnabled = enabled

	if aimlockEnabled then
		notify("Aimlock", "Enabled. Press bind to lock/unlock.")

		safeConnect("AimlockInput", UIS.InputBegan, function(input, gp)
			if gp then return end
			if input.UserInputType ~= aimlockBind and input.KeyCode ~= aimlockBind then return end

			if lockedTarget then
				lockedTarget = nil
				notify("Aimlock", "Unlocked")
				return
			end

			local myTeam = getMyTeamFolder()
			if not myTeam then return end

			local myChar = player.Character
			if not myChar then return end
			local myRoot = myChar:FindFirstChild("HumanoidRootPart")
			if not myRoot then return end

			local closest = nil
			local shortest = math.huge

			local targetFolderName = (myTeam == "ALIVE") and "KILLER" or "ALIVE"
			local targetFolder = workspace:FindFirstChild("PLAYERS") and workspace.PLAYERS:FindFirstChild(targetFolderName)

			if not targetFolder then return end

			for _, other in ipairs(Players:GetPlayers()) do
				if other ~= player and other.Character and other.Character.Parent == targetFolder then
					local root = other.Character:FindFirstChild("HumanoidRootPart")
					if root then
						local dist = (myRoot.Position - root.Position).Magnitude
						if dist < shortest then
							shortest = dist
							closest = root
						end
					end
				end
			end

			if closest then
				lockedTarget = closest
				notify("Aimlock", "Locked onto target")
			end
		end)

		safeConnect("AimlockRender", RunService.RenderStepped, function()
			if lockedTarget and lockedTarget.Parent then
				workspace.CurrentCamera.CFrame = CFrame.new(workspace.CurrentCamera.CFrame.Position, lockedTarget.Position)
			else
				lockedTarget = nil
			end
		end)
	else
		notify("Aimlock", "Disabled.")
		lockedTarget = nil
		disconnect("AimlockInput")
		disconnect("AimlockRender")
	end
end)

--// AUTO PARRY (moved to Combat tab)
createToggle(combatTab, "Auto Parry",false,  function(v)
	autoParryEnabled = v
	getgenv().AutoParryEnabled = v

	if autoParryEnabled then
		if not hasBuffer then
			notify("Auto Parry", "Requires executor with buffer support")
			autoParryEnabled = false
			return
		end

		notify("Auto Parry", "Enabled.")

		autoParryConn = RunService.Heartbeat:Connect(function()
			if not autoParryEnabled then return end

			local char = player.Character
			if not char then return end

			local myRoot = char:FindFirstChild("HumanoidRootPart")
			if not myRoot then return end

			for _, plr in ipairs(Players:GetPlayers()) do
				if plr == player then continue end

				local pchar = plr.Character
				if not pchar then continue end

				local attackerRoot = pchar:FindFirstChild("HumanoidRootPart")
				if not attackerRoot then continue end

				local diff = attackerRoot.Position - myRoot.Position
				local distSq = diff:Dot(diff)

				if distSq > (getgenv().BasicAttackRange or 30)^2 then continue end

				local hum = pchar:FindFirstChildOfClass("Humanoid")
				if not hum then continue end

				local animator = hum:FindFirstChildOfClass("Animator")
				if not animator then continue end

				for _, track in ipairs(animator:GetPlayingAnimationTracks()) do
					if track.IsPlaying then
						local buf = safeBuffer()
						if buf then
							local args = {
								buf.fromstring("\a"),
								buf.fromstring("\254\001\000\254\002\000\006\aAbility\001\002")
							}
							game:GetService("ReplicatedStorage")
								.Modules.Warp.Index.Event.Reliable
								:FireServer(unpack(args))
						end
						break
					end
				end
			end
		end)
	else
		notify("Auto Parry", "Disabled.")
		if autoParryConn then
			autoParryConn:Disconnect()
			autoParryConn = nil
		end
	end
end)

local function newHighlight(obj, color)
	if not obj then return end
	local h = Instance.new("Highlight")
	h.FillColor = color
	h.FillTransparency = 0.5
	h.OutlineColor = color
	h.OutlineTransparency = 0
	h.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
	h.Adornee = obj
	h.Parent = obj
	return h
end

local function createDrawingBox()
	if not hasDrawing then return nil end
	local box = Drawing.new("Square")
	box.Thickness = 2
	box.Filled = false
	box.Transparency = 1
	return box
end

local function updateDrawingBox(box, obj)
	if not box or not obj or not hasDrawing then return end
	local root = obj:FindFirstChild("HumanoidRootPart") or obj.PrimaryPart
	if not root then return end
	local camera = workspace.CurrentCamera
	local vector, onScreen = camera:WorldToViewportPoint(root.Position)
	if onScreen then
		local size = (camera:WorldToViewportPoint(root.Position + Vector3.new(0, 3, 0)) - vector).Y * 2.5
		box.Size = Vector2.new(size * 1.2, size)
		box.Position = Vector2.new(vector.X - box.Size.X / 2, vector.Y - box.Size.Y / 2)
		box.Visible = true
	else
		box.Visible = false
	end
end

local function add(tbl, obj, color)
	if not obj or tbl[obj] then return end

	-- If Drawing not available, force Highlight mode
	local method = espMethod
	if not hasDrawing and (method == "Drawing" or method == "Dual") then
		method = "Highlight"
	end

	if method == "Dual" and hasDrawing then
		tbl[obj] = {
			highlight = newHighlight(obj, color),
			drawing = createDrawingBox()
		}
		if tbl[obj].drawing then
			tbl[obj].drawing.Color = color
		end
	elseif method == "Highlight" then
		tbl[obj] = newHighlight(obj, color)
	elseif hasDrawing then
		local box = createDrawingBox()
		if box then
			box.Color = color
			tbl[obj] = box
		end
	end
end

local function remove(tbl, obj)
	if tbl[obj] then
		if typeof(tbl[obj]) == "table" then
			if tbl[obj].highlight then tbl[obj].highlight:Destroy() end
			if tbl[obj].drawing and hasDrawing then tbl[obj].drawing:Remove() end
		elseif typeof(tbl[obj]) == "Instance" then
			tbl[obj]:Destroy()
		elseif hasDrawing then
			tbl[obj]:Remove()
		end
		tbl[obj] = nil
	end
end

local function clear(tbl)
	for obj, v in pairs(tbl) do
		if v then
			if typeof(v) == "table" then
				if v.highlight then v.highlight:Destroy() end
				if v.drawing and hasDrawing then v.drawing:Remove() end
			elseif typeof(v) == "Instance" then
				v:Destroy()
			elseif hasDrawing then
				v:Remove()
			end
		end
		tbl[obj] = nil
	end
end

-- Fake lag tp
local function stealthMove(targetCFrame)
    local char = player.Character
    local root = char and char:FindFirstChild("HumanoidRootPart")
    if not root then return end

    local startCFrame = root.CFrame
    root.Anchored = true -- Prevent falling/physics flagging during TP

    for i = 1, 3 do
        local alpha = i / 3
        char:PivotTo(startCFrame:Lerp(targetCFrame, alpha))
        game:GetService("RunService").Heartbeat:Wait()
    end

    root.Anchored = false
    root.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
end

local function enterPlatform()
	if safety.active then
		notify("Safety", "Already in platform")
		return
	end

	local char = player.Character
	local root = char and char:FindFirstChild("HumanoidRootPart")
	if not root then return end

	-- cleanup any leftover platform FIRST
	if safety.part then
		safety.part:Destroy()
		safety.part = nil
	end

	-- save position BEFORE moving
	safety.lastCFrame = root.CFrame

	local pos = root.Position
	local platformPos = pos - Vector3.new(0, 100, 0)

	-- create platform
	safety.part = Instance.new("Part")
	safety.part.Size = Vector3.new(50, 1, 50)
	safety.part.Anchored = true
	safety.part.Position = platformPos
	safety.part.Name = "SafetyPlate"
	safety.part.Parent = workspace

	-- now mark active
	safety.active = true
	-- move player
	task.spawn(function()
		stealthMove(CFrame.new(platformPos + Vector3.new(0, 5, 0)))
	end)
end

-- EXIT PLATFORM
local function exitPlatform()
	if not safety.active then
		notify("Safety", "Not in platform")
		return
	end

	if not safety.lastCFrame then
		notify("Safety", "No saved position")
		return
	end

	if safety.moving then return end
	safety.moving = true

	local _exit_lastCF = safety.lastCFrame
	local _exit_killerRoot = getKillerRoot()
	local _exit_dist = _exit_killerRoot and (_exit_lastCF.Position - _exit_killerRoot.Position).Magnitude or math.huge

	-- SAFETY CHECK
	if _exit_dist < 40 then
		notify("Safety", "Killer is " .. math.floor(_exit_dist) .. " studs from return position")
		safety.moving = false
		return
	end

	notify("Safety", "Returning to position")

	task.spawn(function()
		stealthMove(_exit_lastCF)

		-- CLEANUP AFTER MOVE
		if safety.part then
			safety.part:Destroy()
			safety.part = nil
		end

		safety.active = false
		safety.lastCFrame = nil
		safety.moving = false
	end)
end

-- AUTO ENTER (ANTI-DEATH)
safeConnect("AutoSafety", RunService.Heartbeat, function()
	if not safety.auto then return end
	if safety.active then return end 

	local char = player.Character
	local hum = char and char:FindFirstChildOfClass("Humanoid")

	if not hum or hum.Health <= 0 then return end
	if not inMatch() then return end

	local now = tick()

	if hum.Health < safety.threshold and (now - safety.lastTrigger >= safety.cooldown) then
		safety.lastTrigger = now
		enterPlatform() 
		notify("Safety", "Low HP detected")
	end
end)

-- Lerp teleport function
local tpSpeed = 25
local lerping = false
local currentTarget = nil

local function lerpTo(targetCFrame)
	local char = player.Character or player.CharacterAdded:Wait()
	local root = char:FindFirstChild("HumanoidRootPart")
	if not root then return end
	local targetPos = targetCFrame.Position

	if currentTarget and (currentTarget - targetPos).Magnitude < 0.5 then
		lerping = false
		currentTarget = nil
		return
	end

	currentTarget = targetPos
	lerping = true

	while lerping and root.Parent do
		local currentPos = root.Position
		local direction = (targetPos - currentPos)
		local distance = direction.Magnitude

		if distance <= 0.5 then
			break
		end

		local step = math.min(tpSpeed * RunService.RenderStepped:Wait(), distance)
		root.CFrame = CFrame.new(currentPos + direction.Unit * step, targetPos)
	end

	if lerping and root.Parent then
		root.CFrame = CFrame.new(targetPos)
	end

	lerping = false
	currentTarget = nil
end

local function TeleportTo(model)
	if not model then return end
	local part = model.PrimaryPart or model:FindFirstChildWhichIsA("BasePart")
	if not part then return end
	local frontCFrame = part.CFrame * CFrame.new(0, 0, -5)
	lerpTo(frontCFrame)
end

local function getOrderedGenerators()
	local mapsFolder = workspace:FindFirstChild("MAPS")
	if not mapsFolder then return {} end
	local gameMap = mapsFolder:FindFirstChild("GAME MAP")
	if not gameMap then return {} end
	local generators = gameMap:FindFirstChild("Generators")
	if not generators then return {} end
	local models = {}
	for _, v in ipairs(generators:GetChildren()) do
		if v:IsA("Model") then table.insert(models, v) end
	end
	table.sort(models, function(a, b)
		local ao = a:GetAttribute("Order") or 0
		local bo = b:GetAttribute("Order") or 0
		return ao < bo
	end)
	return models
end

--.// Main drag
local dragging = false
local dragStart, startPos

MainFrame.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
		dragging = true
		dragStart = input.Position
		startPos = MainFrame.Position
		
		-- Use a one-time connection to stop dragging when the mouse is released anywhere
		local releaseConn
		releaseConn = UIS.InputEnded:Connect(function(EndInput)
			if EndInput.UserInputType == Enum.UserInputType.MouseButton1 or EndInput.UserInputType == Enum.UserInputType.Touch then
				dragging = false
				releaseConn:Disconnect()
			end
		end)
	end
end)

-- Track movement globally so the mouse doesn't "outrun" the UI
UIS.InputChanged:Connect(function(input)
	if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
		local delta = input.Position - dragStart
		MainFrame.Position = UDim2.new(
			startPos.X.Scale, startPos.X.Offset + delta.X,
			startPos.Y.Scale, startPos.Y.Offset + delta.Y
		)
	end
end)


UIS.InputBegan:Connect(function(input, gameProcessed)
	-- Ignore chat / textboxes
	if gameProcessed or UIS:GetFocusedTextBox() then return end

	if input.KeyCode == Enum.KeyCode.RightAlt then
		if MainFrame.Visible then
			-- HIDE UI
			MainFrame.Visible = false
			reopenBtn.Visible = true
			notify("UI", "Hidden (Right Alt to reopen)")
		else
			-- SHOW UI
			MainFrame.Visible = true
			reopenBtn.Visible = false
			notify("UI", "Opened")
		end
	end
end)
